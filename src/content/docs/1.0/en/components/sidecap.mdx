        const _add_event_listeners = (sidecap) => {

            })
        }

        const _init = () => {
          _add_event_listeners_dom();

          [].forEach.call(elements, (sidecap) => {
            _add_event_listeners(sidecap);
          })
        };

        // Initiating the accordion
        if(sidecap) {
          _init();
        }

        return ({
          elements : elements,
          open : () => {}
        })
      }
